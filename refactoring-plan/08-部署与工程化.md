## 1. 开发环境

### 1.1 前置依赖

| 工具 | 版本 | 用途 |
|------|------|------|
| Python | >= 3.12 | 后端运行时 |
| Node.js | >= 20 LTS | 前端构建 |
| pnpm | >= 9 | 前端包管理 |
| PostgreSQL | >= 16 | 数据库 |

| Docker Compose | >= 2.24 | 编排 |

### 1.2 快速启动

```bash
# 1. 克隆仓库
git clone https://github.com/saltflow/songHut
cd songHut

# 2. 后端
cd backend
python -m venv .venv
source .venv/bin/activate  # Windows: .venv\Scripts\activate
pip install -e ".[dev]"
cp .env.example .env        # 编辑配置
alembic upgrade head        # 数据库迁移
uvicorn app.main:app --reload --port 8000

# 3. 前端
cd ../frontend
pnpm install
pnpm dev:web                # Web 开发
# 或
pnpm dev:desktop            # Electron 开发

# 4. Worker (新开终端)
cd backend
celery -A app.workers worker -l info

# 5. 浏览器打开 http://localhost:5173
```

### 1.4 开发 Makefile

```makefile
# Makefile
.PHONY: dev dev-backend dev-frontend dev-worker db-migrate lint test clean

# 开发 — 依赖 Docker
dev:
	docker compose -f docker-compose.dev.yml up --build

# 只启动依赖 (postgres + redis)
dev-deps:
	docker compose -f docker-compose.dev.yml up postgres redis -d

# 后端本地开发 (需要先启动 deps)
dev-backend: dev-deps
	cd backend && uv run uvicorn app.main:app --reload --port 8000

# 前端本地开发
dev-frontend:
	cd frontend && pnpm dev:web

# Celery Worker
dev-worker: dev-deps
	cd backend && celery -A app.workers worker -l info

# 数据库迁移
db-migrate:
	cd backend && alembic upgrade head

db-make-migrations:
	cd backend && alembic revision --autogenerate -m "$(message)"

# 代码规范检查
lint:
	cd backend && ruff check && mypy app
	cd frontend && pnpm lint && pnpm typecheck

# 测试
test:
	cd backend && pytest -v --cov=app
	cd frontend && pnpm test

# 清理
clean:
	find . -type d -name "__pycache__" -exec rm -rf {} + 2>/dev/null || true
	find . -type d -name ".ruff_cache" -exec rm -rf {} + 2>/dev/null || true
	find . -type d -name "node_modules" -exec rm -rf {} + 2>/dev/null || true
	rm -rf backend/.venv
	rm -rf frontend/packages/*/dist
	rm -rf frontend/packages/app/release

# 安装
install:
	cd backend && pip install -e ".[dev]"
	cd frontend && pnpm install

# 构建
build:
	cd frontend && pnpm build:web && pnpm build:desktop
```

    default upgrade;

    listen 80;

    server_name api.songhut.com;

    ssl_certificate_key /etc/nginx/ssl/key.pem;

    # API 反向代理
    location /api/ {
        proxy_pass http://backend;
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;
        proxy_read_timeout 120s;
    }

    # WebSocket
    location /ws/ {
        proxy_pass http://backend;
        proxy_http_version 1.1;
        proxy_set_header Upgrade $http_upgrade;
        proxy_set_header Connection $connection_upgrade;
        proxy_set_header Host $host;
        proxy_read_timeout 86400s;
    }

    # 静态文件 (直接由 Nginx 提供避免 Python 负担)
    location /files/ {
        alias /data/;
        internal;  # 通过 X-Accel-Redirect 由后端控制权限
    }

    # Swagger UI
    location /docs {
        proxy_pass http://backend;
    }

    location /openapi.json {
        proxy_pass http://backend;
    }
}

    listen 443 ssl http2;
    server_name app.songhut.com;

    ssl_certificate /etc/nginx/ssl/cert.pem;
    ssl_certificate_key /etc/nginx/ssl/key.pem;

    root /usr/share/nginx/html;
    index index.html;

    # SPA: 所有路由 fallback 到 index.html

        try_files $uri $uri/ /index.html;

## 5. 数据库迁移 (Alembic)

```python
# backend/alembic/env.py
from logging.config import fileConfig
from sqlalchemy import engine_from_config, pool
from alembic import context
from app.models.base import Base
from app.core.config import settings

config = context.config
config.set_main_option("sqlalchemy.url", settings.database_url)

target_metadata = Base.metadata


def run_migrations_online():
    connectable = engine_from_config(
        config.get_section(config.config_ini_section),
        prefix="sqlalchemy.",
        poolclass=pool.NullPool,
    )
    with connectable.connect() as connection:
        context.configure(
            connection=connection,
            target_metadata=target_metadata,
        )
        with context.begin_transaction():
            context.run_migrations()
```

### 8.2 迁移清单

| 项目 | 迁移策略 |
|------|---------|
| 用户 | 创建新用户表，密码设置为无效 (需重置) |
| 项目 | Repository → Project，owner 对应 |
| 文件 | 复制磁盘文件到新存储，重建 DB 记录 |
| 任务 | 旧任务不迁移 (无状态)，用户需重新提交 |
| 注释/评论 | 弃用 (旧系统从未使用过) |
| 音频/MIDI | 保留原始文件，格式不变 |
