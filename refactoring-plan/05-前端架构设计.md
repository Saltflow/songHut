# SongHut 2.0 — 前端架构设计

## 1. 设计原则

1. **功能式状态管理**：状态变换用 `纯 reducer` + Zustand。**禁止**在组件内直接 `setState` 修改深层嵌套。
2. **单向数据流**：`API → Store → Selector → Component`。**禁止**组件直接写 Store（通过 action）。
3. **跨端意识**：95% 代码在 `shared/` 中，平台差异通过 `platform.ts` 抽象层解决。
4. **组件叶子化**：Page 组合 Feature，Feature 组合通用 UI 组件。组件尽可能无状态。
5. **类型安全**：API 响应类型自动从后端 OpenAPI 生成（openapi-typescript）。
6. **React 18 Concurrent**：使用 `Suspense`、`useTransition`、`useDeferredValue` 处理异步 UI。

### 3.1 Axios 实例

```typescript
// shared/src/api/client.ts
import axios, { AxiosError, InternalAxiosRequestConfig } from 'axios';
import { useAuthStore } from '../stores/auth.store';

const API_BASE = import.meta.env.VITE_API_BASE || 'http://localhost:8000/api/v1';

export const apiClient = axios.create({
  baseURL: API_BASE,
  timeout: 30_000,
  headers: { 'Content-Type': 'application/json' },
});

// ==== 请求拦截器 — 自动注入 token ====
apiClient.interceptors.request.use((config: InternalAxiosRequestConfig) => {
  const token = useAuthStore.getState().accessToken;
  if (token) {
    config.headers.Authorization = `Bearer ${token}`;
  }
  return config;
});

// ==== 响应拦截器 — 统一错误处理 + token 自动刷新 ====
apiClient.interceptors.response.use(
  (response) => response,
  async (error: AxiosError<ApiErrorResponse>) => {
    const originalRequest = error.config as any;

    // 401 → 尝试刷新 token
    if (error.response?.status === 401 && !originalRequest._retry) {
      originalRequest._retry = true;
      const refreshed = await useAuthStore.getState().refreshToken();
      if (refreshed) {
        originalRequest.headers.Authorization =
          `Bearer ${useAuthStore.getState().accessToken}`;
        return apiClient(originalRequest);
      }
      // 刷新失败 → 强制登出
      useAuthStore.getState().logout();
    }

    return Promise.reject(normalizeError(error));
  }
);

// ==== 统一响应类型 ====
export interface ApiResponse<T> {
  ok: true;
  data: T;
}

export interface ApiError {
  ok: false;
  error: {
    code: string;
    message: string;
    details?: Record<string, unknown>;
  };
}

export type ApiResult<T> = ApiResponse<T> | ApiError;

export interface PageResponse<T> {
  items: T[];
  total: number;
  page: number;
  page_size: number;
  total_pages: number;
}
```

// ==== 类型 ====

  id: string;

  nickname: string;
  email: string | null;

  password: string;

    apiClient.post('/auth/logout').then(r => r.data),

  sendSms: (phone: string): Promise<ApiResult<{ expires_in: number }>> =>
    apiClient.post('/auth/send-sms', { phone }).then(r => r.data),

import { apiClient, ApiResult, UploadProgressCallback } from './client';

export interface FileDto {
  id: string;
  owner_id: string;
  project_id: string | null;
  filename: string;
  category: FileCategory;
  mime_type: string;
  file_size: number;
  storage_path: string;
  duration_ms: number | null;
  metadata: Record<string, unknown>;
  created_at: string;
}

export type FileCategory =
  | 'recording' | 'melody' | 'accompaniment'
  | 'vocal' | 'lyrics' | 'score' | 'image' | 'other';

export const filesApi = {
  upload: (
    projectId: string,
    file: File,
    category: FileCategory,
    onProgress?: UploadProgressCallback,  // (loaded, total) => void
  ): Promise<ApiResult<FileDto>> => {
    const form = new FormData();
    form.append('file', file);
    form.append('category', category);
    return apiClient.post(`/projects/${projectId}/files`, form, {
      headers: { 'Content-Type': 'multipart/form-data' },
      onUploadProgress: onProgress
        ? (e) => onProgress(e.loaded, e.total ?? e.loaded)
        : undefined,
    }).then(r => r.data);
  },

  download: (fileId: string): Promise<Blob> =>
    apiClient.get(`/files/${fileId}/download`, {
      responseType: 'blob',
    }).then(r => r.data),

  get: (fileId: string): Promise<ApiResult<FileDto>> =>
    apiClient.get(`/files/${fileId}`).then(r => r.data),

  delete: (fileId: string): Promise<ApiResult<null>> =>
    apiClient.delete(`/files/${fileId}`).then(r => r.data),

  setFeatured: (fileId: string): Promise<ApiResult<null>> =>
    apiClient.post(`/files/${fileId}/feature`).then(r => r.data),
};

## 4. 状态管理层 (Zustand)

```typescript
// shared/src/stores/auth.store.ts
import { create } from 'zustand';
import { persist } from 'zustand/middleware';
import { authApi, UserDto, AuthTokens } from '../api/auth';

// ==== 纯 reducer — 无副作用的类型接口 ====
interface AuthState {
  // 数据
  user: UserDto | null;
  accessToken: string | null;
  refreshToken: string | null;
  isAuthenticated: boolean;

  // Action creators (副作用在内部处理，对外暴露纯接口)
  login: (phone: string, password: string) => Promise<boolean>;
  register: (data: RegisterRequest) => Promise<boolean>;
  logout: () => void;
  refreshToken: () => Promise<boolean>;
  setUser: (user: UserDto) => void;
}

export const useAuthStore = create<AuthState>()(
  persist(
    (set, get) => ({
      user: null,
      accessToken: null,
      refreshToken: null,
      isAuthenticated: false,

      login: async (phone, password) => {
        const result = await authApi.login({ phone, password });
        if (result.ok) {
          set({
            user: result.data.user,
            accessToken: result.data.access_token,
            refreshToken: result.data.refresh_token,
            isAuthenticated: true,
          });
          return true;
        }
        return false;
      },

      register: async (data) => {
        const result = await authApi.register(data);
        if (result.ok) {
          set({
            user: result.data.user,
            accessToken: result.data.access_token,
            refreshToken: result.data.refresh_token,
            isAuthenticated: true,
          });
          return true;
        }
        return false;
      },

      logout: () => {
        authApi.logout().catch(() => {}); // fire-and-forget
        set({
          user: null,
          accessToken: null,
          refreshToken: null,
          isAuthenticated: false,
        });
      },

      refreshToken: async () => {
        const currentRefresh = get().refreshToken;
        if (!currentRefresh) return false;

        const result = await authApi.refresh(currentRefresh);
        if (result.ok) {
          set({
            accessToken: result.data.access_token,
            refreshToken: result.data.refresh_token,
          });
          return true;
        }
        return false;
      },

      setUser: (user) => set({ user }),
    }),
    {
      name: 'songhut-auth',
      partialize: (state) => ({
        accessToken: state.accessToken,
        refreshToken: state.refreshToken,
        user: state.user,
        isAuthenticated: state.isAuthenticated,
      }),
    }
  )
);
```

import { create } from 'zustand';
import { projectsApi, ProjectDto, ProjectDetailDto, CreateProjectRequest } from '../api/projects';

interface ProjectState {
  // 数据
  projects: ProjectDto[];
  currentProject: ProjectDetailDto | null;
  isLoading: boolean;

  // Actions — 纯函数接口
  fetchProjects: () => Promise<void>;
  fetchProjectDetail: (id: string) => Promise<void>;
  createProject: (data: CreateProjectRequest) => Promise<boolean>;
}

export const useProjectStore = create<ProjectState>()((set, get) => ({
  projects: [],
  currentProject: null,
  isLoading: false,

  fetchProjects: async () => {
    set({ isLoading: true });
    const result = await projectsApi.list({ page: 1, page_size: 50 });
    if (result.ok) {
      set({ projects: result.data.items, isLoading: false });
    } else {
      set({ isLoading: false });
    }
  },

  fetchProjectDetail: async (id) => {
    set({ isLoading: true });
    const result = await projectsApi.getDetail(id);
    if (result.ok) {
      set({ currentProject: result.data, isLoading: false });
    } else {
      set({ isLoading: false });
    }
  },

  createProject: async (data) => {
    const result = await projectsApi.create(data);
    if (result.ok) {
      // 乐观更新: 追加到列表
      set((s) => ({ projects: [result.data, ...s.projects] }));
      return true;
    }
    return false;
  },
}));

// shared/src/hooks/useAudioRecorder.ts
import { useState, useRef, useCallback } from 'react';

interface AudioRecorderState {
  isRecording: boolean;
  durationMs: number;
  audioBlob: Blob | null;
  error: string | null;
}

export function useAudioRecorder() {
  const [state, setState] = useState<AudioRecorderState>({
    isRecording: false,
    durationMs: 0,
    audioBlob: null,
    error: null,
  });

  const mediaRecorderRef = useRef<MediaRecorder | null>(null);
  const chunksRef = useRef<Blob[]>([]);
  const timerRef = useRef<number>(0);

  const startRecording = useCallback(async () => {
    try {
      const stream = await navigator.mediaDevices.getUserMedia({ audio: true });
      const recorder = new MediaRecorder(stream, {
        mimeType: MediaRecorder.isTypeSupported('audio/webm;codecs=opus')
          ? 'audio/webm;codecs=opus'
          : 'audio/webm',
      });

      chunksRef.current = [];
      recorder.ondataavailable = (e) => {
        if (e.data.size > 0) chunksRef.current.push(e.data);
      };

      recorder.onstop = () => {
        const blob = new Blob(chunksRef.current, { type: recorder.mimeType });
        stream.getTracks().forEach((t) => t.stop());
        setState((s) => ({ ...s, isRecording: false, audioBlob: blob }));
      };

      recorder.start(100); // 每 100ms 收集一次数据
      mediaRecorderRef.current = recorder;

      // 计时
      const startTime = Date.now();
      timerRef.current = window.setInterval(() => {
        setState((s) => ({ ...s, durationMs: Date.now() - startTime }));
      }, 100);

      setState((s) => ({ ...s, isRecording: true, error: null }));
    } catch (e: any) {
      setState((s) => ({ ...s, error: e.message }));
    }
  }, []);

  const stopRecording = useCallback(() => {
    mediaRecorderRef.current?.stop();
    clearInterval(timerRef.current);
  }, []);

  return {
    ...state,
    startRecording,
    stopRecording,
    reset: () => setState({
      isRecording: false, durationMs: 0, audioBlob: null, error: null,
    }),
  };
}

// shared/src/hooks/useFileUpload.ts
import { useState, useCallback } from 'react';
import { filesApi, FileDto, FileCategory } from '../api/files';

interface UploadState {
  isUploading: boolean;
  progress: number;  // 0-100
  file: FileDto | null;
  error: string | null;
}

export function useFileUpload() {
  const [state, setState] = useState<UploadState>({
    isUploading: false,
    progress: 0,
    file: null,
    error: null,
  });

  const upload = useCallback(
    async (projectId: string, file: File, category: FileCategory) => {
      setState({ isUploading: true, progress: 0, file: null, error: null });

      const result = await filesApi.upload(projectId, file, category,
        (loaded, total) => {
          setState((s) => ({ ...s, progress: Math.round((loaded / total) * 100) }));
        },
      );

      if (result.ok) {
        setState((s) => ({ ...s, isUploading: false, file: result.data }));
      } else {
        setState((s) => ({
          ...s, isUploading: false,
          error: result.error.message,
        }));
      }
    },
    [],
  );

  return { ...state, upload, reset: () => setState({
    isUploading: false, progress: 0, file: null, error: null,
  }) };
}

// shared/src/hooks/useTaskWebSocket.ts
import { useEffect, useRef, useCallback } from 'react';
import { useAuthStore } from '../stores/auth.store';

type TaskEvent =
  | { event: 'task:progress'; data: { task_id: string; status: string; progress: number; message?: string } }
  | { event: 'task:completed'; data: { task_id: string; result_file_id: string } }
  | { event: 'task:failed'; data: { task_id: string; error_message: string } }
  | { event: 'ping' };

export function useTaskWebSocket(
  taskId: string,
  onEvent: (event: TaskEvent) => void,
) {
  const wsRef = useRef<WebSocket | null>(null);
  const token = useAuthStore((s) => s.accessToken);

  useEffect(() => {
    if (!token) return;

    const protocol = window.location.protocol === 'https:' ? 'wss:' : 'ws:';
    const host = import.meta.env.VITE_WS_HOST || 'localhost:8000';
    const ws = new WebSocket(`${protocol}//${host}/api/v1/ws/tasks/${taskId}`);

    ws.onopen = () => {
      ws.send(JSON.stringify({ type: 'auth', token: `Bearer ${token}` }));
    };

    ws.onmessage = (msg) => {
      const event: TaskEvent = JSON.parse(msg.data);
      if (event.event === 'ping') {
        ws.send(JSON.stringify({ event: 'pong' }));
        return;
      }
      onEvent(event);
    };

    ws.onerror = () => {
      // 重连逻辑由外层 poll 兜底
    };

    wsRef.current = ws;

    return () => {
      ws.close();
    };
  }, [taskId, token]);
}

// shared/src/lib/platform.ts

export interface PlatformAPI {
  /** 是否在 Electron 中运行 */
  isElectron: boolean;

  /** 是否在移动浏览器中运行 */
  isMobile: boolean;

  /** 打开系统文件选择对话框 (Electron: 原生对话框, Web: <input type=file>) */
  openFileDialog(options?: {
    accept?: string;
    multiple?: boolean;
  }): Promise<File[] | null>;

  /** 读取文件内容 (Electron: fs.readFile, Web: FileReader) */
  readFile(path: string): Promise<ArrayBuffer>;

  /** 保存文件到磁盘 (Electron: 对话框 + fs.writeFile, Web: download link) */
  saveFile(data: Blob, filename: string): Promise<void>;

  /** 获取平台信息 */
  platform: 'electron' | 'web' | 'mobile';

  /** 全局快捷键注册 (仅 Electron) */
  registerShortcut?(key: string, handler: () => void): () => void;
}

// Web 实现
const webPlatform: PlatformAPI = {
  isElectron: false,
  isMobile: /Android|iPhone|iPad/i.test(navigator.userAgent),

  openFileDialog: async ({ accept, multiple } = {}) => {
    return new Promise((resolve) => {
      const input = document.createElement('input');
      input.type = 'file';
      if (accept) input.accept = accept;
      input.multiple = multiple ?? false;
      input.onchange = () => resolve(Array.from(input.files ?? []));
      input.click();
    });
  },

  readFile: async (path) => {
    // Web 不支持任意路径读取 — 通过 API 下载
    throw new Error('Web platform cannot read arbitrary file paths');
  },

  saveFile: async (data, filename) => {
    const url = URL.createObjectURL(data);
    const a = document.createElement('a');
    a.href = url;
    a.download = filename;
    a.click();
    URL.revokeObjectURL(url);
  },

  platform: 'web',
};

// Electron 实现
const electronPlatform: PlatformAPI = {
  isElectron: true,
  isMobile: false,

  openFileDialog: async (options) => {
    // Electron IPC: 主进程打开对话框
    const files = await window.electronAPI.openFileDialog(options);
    return files ?? null;
  },

  readFile: async (path) => {
    const buffer = await window.electronAPI.readFile(path);
    return buffer;
  },

  saveFile: async (data, filename) => {
    await window.electronAPI.saveFile(data, filename);
  },

  platform: 'electron',

  registerShortcut: (key, handler) => {
    window.electronAPI.registerShortcut(key, handler);
    return () => window.electronAPI.unregisterShortcut(key, handler);
  },
};

// 自动选择平台实现
export const platform: PlatformAPI =
  typeof window !== 'undefined' && window.electronAPI
    ? electronPlatform
    : webPlatform;
