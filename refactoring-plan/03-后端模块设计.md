## 1. 模块依赖图

```
                    ┌──────────────┐
                    │   API 路由层   │  (FastAPI routers — 薄，仅编排)
                    └──────┬───────┘
                           │ 调用
                    ┌──────▼───────┐
                    │  Service 层   │  (纯函数业务逻辑)
                    └──┬───┬───┬───┘
                       │   │   │
          ┌────────────┘   │   └──────────────┐
          ▼                ▼                  ▼
   ┌────────────┐   ┌───────────┐   ┌─────────────────┐
   │  Repository │   │  Storage  │   │  Algorithm       │
   │  (DB 操作)  │   │  (文件)   │   │  (纯数据变换)     │
   └────────────┘   └───────────┘   └─────────────────┘
          │                                              │
          ▼                                              ▼
   ┌────────────┐                                ┌─────────────────┐
   │ PostgreSQL │                                │  Celery Worker  │
   └────────────┘                                └─────────────────┘
```

**层层依赖规则：**
- 路由层 → Service 层 → Repository / Storage / Algorithm
- Service 层**不直接**依赖路由层或 FastAPI
- Service 层只操作 DTO，**不操作** ORM 对象
- Algorithm 层**无任何** I/O 或架构依赖

### 2.2 result.py — Result 代数类型

```python
# app/core/result.py
from __future__ import annotations
from dataclasses import dataclass
from typing import Generic, TypeVar, Union

from app.core.errors import AppError

T = TypeVar("T")
E = TypeVar("E")


@dataclass(frozen=True)
class Ok(Generic[T]):
    """成功结果"""
    value: T


@dataclass(frozen=True)
class Err(Generic[E]):
    """失败结果，携带错误信息"""
    error: E


Result = Union[Ok[T], Err[E]]


# ==== 组合子 (Combinators) ====

def map_ok(result: Result[T, E], fn):
    """OK 时变换值，Err 时透传"""
    match result:
        case Ok(value):
            return Ok(fn(value))
        case Err() as err:
            return err


def map_err(result: Result[T, E], fn):
    """Err 时变换错误，OK 时透传"""
    match result:
        case Ok() as ok:
            return ok
        case Err(error):
            return Err(fn(error))


def and_then(result: Result[T, E], fn):
    """Monadic bind — 当前 Result OK 时执行 fn，Err 时短路"""
    match result:
        case Ok(value):
            return fn(value)
        case Err() as err:
            return err


def is_ok(result: Result[T, E]) -> bool:
    """类型守卫"""
    return isinstance(result, Ok)


def unwrap(result: Result[T, E]) -> T:
    """不安全：仅在确信为 OK 时使用"""
    match result:
        case Ok(value):
            return value
        case Err(error):
            raise RuntimeError(f"unwrap on Err: {error}")


# ==== Pipeline 操作符 ====

def pipe(value, *fns):
    """函数式流水线: pipe(x, f, g, h) = h(g(f(x)))"""
    for fn in fns:
        value = fn(value)
    return value

# 用法示例:
# pipe(
#     create_user(db, dto),            # Result[User, AppError]
#     and_then(lambda u: send_welcome(u)),  # Result[User, AppError]
#     and_then(lambda u: create_project(db, u.id, "默认项目")),  # Result[Project, AppError]
# )
```

    details: dict[str, Any] | None = None

# ==== 预置工厂函数 ====

def invalid_credentials(msg: str | None = None) -> AppError:
    return AppError(
        code="AUTH_INVALID_CREDENTIALS",
        message=msg or "手机号或密码错误",
        http_status=401,
    )


def not_found(entity: str, id: str) -> AppError:

### 2.5 security.py — JWT 和密码

```python
# app/core/security.py
from __future__ import annotations
from datetime import datetime, timedelta, timezone
from uuid import UUID

from jose import JWTError, jwt
from passlib.context import CryptContext

from app.core.config import settings
from app.core.result import Result, Ok, Err, AppError

# bcrypt 上下文
pwd_context = CryptContext(schemes=["bcrypt"], deprecated="auto")


# ======== 密码 (纯函数) ========

def hash_password(raw: str) -> str:
    """不可逆哈希"""
    return pwd_context.hash(raw)


def verify_password(raw: str, hashed: str) -> bool:
    """验证明文 vs 哈希"""
    return pwd_context.verify(raw, hashed)


# ======== JWT 令牌 (纯函数) ========

def _now() -> datetime:
    return datetime.now(timezone.utc)


def create_access_token(user_id: UUID) -> str:
    """生成 access token (短有效期)"""
    expire = _now() + timedelta(minutes=settings.jwt_access_expire_minutes)
    payload = {
        "sub": str(user_id),
        "exp": expire,
        "type": "access",
    }
    return jwt.encode(payload, settings.jwt_secret, algorithm=settings.jwt_algorithm)


def create_refresh_token(user_id: UUID) -> str:
    """生成 refresh token (长有效期, 支持轮换)"""
    expire = _now() + timedelta(days=settings.jwt_refresh_expire_days)
    payload = {
        "sub": str(user_id),
        "exp": expire,
        "type": "refresh",
    }
    return jwt.encode(payload, settings.jwt_secret, algorithm=settings.jwt_algorithm)


def verify_access_token(token: str) -> Result[UUID, AppError]:
    """验证 access token → 返回 user_id"""
    try:
        payload = jwt.decode(
            token, settings.jwt_secret,
            algorithms=[settings.jwt_algorithm],
        )
        if payload.get("type") != "access":
            return Err(AppError(code="TOKEN_INVALID", message="不是 access token",
                                http_status=401))
        user_id = UUID(payload["sub"])
        return Ok(user_id)
    except JWTError:
        return Err(AppError(code="TOKEN_EXPIRED", message="token 过期或无效",
                            http_status=401))


def verify_refresh_token(token: str) -> Result[UUID, AppError]:
    """验证 refresh token。成功时返回 user_id。"""
    try:
        payload = jwt.decode(
            token, settings.jwt_secret,
            algorithms=[settings.jwt_algorithm],
        )
        if payload.get("type") != "refresh":
            return Err(AppError(code="TOKEN_INVALID", message="不是 refresh token",
                                http_status=401))
        user_id = UUID(payload["sub"])
        return Ok(user_id)
    except JWTError:
        return Err(AppError(code="TOKEN_INVALID", message="refresh token 无效或过期",
                            http_status=401))
```

### 2.6 deps.py — 依赖注入

```python
# app/core/deps.py
from uuid import UUID
from fastapi import Depends, HTTPException, Header
from sqlalchemy.ext.asyncio import AsyncSession

from app.core.database import get_db
from app.core.security import verify_access_token
from app.core.result import is_ok, unwrap


async def get_current_user_id(
    authorization: str = Header(...),
) -> UUID:
    """从 Authorization header 提取当前用户 ID。
    给所有需要认证的路由使用。"""
    scheme, _, token = authorization.partition(" ")
    if scheme.lower() != "bearer" or not token:
        raise HTTPException(status_code=401, detail="无效的 Authorization header")

    result = verify_access_token(token)
    if not is_ok(result):
        raise HTTPException(
            status_code=401,
            detail=unwrap(result).error.message,
        )
    return unwrap(result).value


async def get_db_session() -> AsyncSession:
    """数据库 session (yield 给路由)"""
    async with get_db() as session:
        yield session
```

class LocalStorage:
    """本地文件系统实现"""

    def __init__(self, base_path: str):
        self.base_path = Path(base_path)

    async def save(self, path: str, data: bytes) -> Result[str, AppError]:
        full_path = self.base_path / path
        full_path.parent.mkdir(parents=True, exist_ok=True)
        try:
            full_path.write_bytes(data)
            return Ok(str(full_path))

### 8.2 Pipeline 编排器

```python
# app/algorithms/pipeline.py
from app.algorithms.pitch.protocol import PitchDetector
from app.algorithms.beat.protocol import BeatTracker
from app.algorithms.midi.quantize import Quantizer
from app.algorithms.midi.generator import MidiGenerator
from app.algorithms.chord.protocol import ChordDetector
from app.algorithms.accompany.drum import DrumGenerator
from app.algorithms.accompany.bass import BassGenerator
from app.algorithms.accompany.chord_accompany import ChordAccompaniment

from app.schemas.common import MelodyParams
from app.core.audio import AudioBuffer


def humming_to_melody(
    audio: AudioBuffer,
    params: MelodyParams,
    pitch_detector: PitchDetector,
    beat_tracker: BeatTracker,
    quantizer: Quantizer,
    midi_generator: MidiGenerator,
    chord_detector: ChordDetector,
    drum_gen: DrumGenerator,
    bass_gen: BassGenerator,
    chord_accomp: ChordAccompaniment,
) -> MidiOutput:
    """完整的哼唱转旋律 pipeline。纯函数 — 所有依赖显式传入。"""

    # Step 1: 音高检测
    frames = pitch_detector.detect(audio)

    # Step 2: 节拍跟踪
    beats = beat_tracker.detect(audio)

    # Step 3: 音符量化
    notes = quantizer.quantize(frames, beats)

    # Step 4: 生成主旋律 MIDI 轨道
    melody_track = midi_generator.generate(notes, instrument=params.instrument)

    tracks = [melody_track]

    # Step 5: 和弦分析 (如果需要伴奏)
    chords = chord_detector.detect(notes) if (params.is_chord or params.is_bass) else []

    # Step 6: 生成伴奏 (可选)
    if params.is_drum:
        drum_track = drum_gen.generate(beats)
        tracks.append(drum_track)

    if params.is_bass and chords:
        bass_track = bass_gen.generate(chords, beats)
        tracks.append(bass_track)

    if params.is_chord and chords:
        chord_track = chord_accomp.generate(chords, beats, params.chord_style)
        tracks.append(chord_track)

    # Step 7: 合并
    return MidiOutput(tracks=tracks, tempo_bpm=beats.bpm)
```
