# SongHut 2.0 — 乐谱渲染与可视化方案

## 1. 总体架构

```
MIDI File (算法产出)
    │
    ▼
[后端] ScoreConverter
    │  music21 / partitura → MusicXML
    │
    ▼
[后端] 存储: scores 表 (musicxml + vexflow_json)
    │
    ▼
[前端] 请求 GET /api/v1/scores/{id}
    │  返回 ScoreDto (含 musicxml 或 vexflow_json)
    │
    ▼
[前端] VexFlow 5 渲染 Canvas/SVG
    │
    ├── 静态乐谱展示
    ├── 播放跟随高亮
    ├── 编辑/移调
    └── 导出 MusicXML / PDF / MIDI / SVG
```

## 2. 技术选型

| 组件 | 用途 | 理由 |
|------|------|------|
| **music21** (Python) | MIDI → MusicXML | 学术级乐理分析库，支持调号/拍号推算 |
| **VexFlow 5** (TypeScript) | 前端 Canvas 渲染 | 轻量、无外部字体依赖、React 友好 |
| **OpenSheetMusicDisplay** (备选) | 复杂 MusicXML 渲染 | 功能更全但体重大 (~1MB) |
| **html2canvas** + **jsPDF** | 导出 PDF | 浏览器端完成，无需后端 |

## 3. 后端乐谱生成

### 3.1 MIDI → MusicXML 转换

```python
# app/algorithms/score/converter.py
from music21 import converter, stream, note, chord, tempo, meter, key
import tempfile


class Music21ScoreConverter:
    """使用 music21 将 MIDI 转换为 MusicXML。"""

    def midi_to_musicxml(self, midi_bytes: bytes) -> str:
        """MIDI 二进制 → MusicXML 字符串。纯函数。"""
        with tempfile.NamedTemporaryFile(suffix=".mid", delete=False) as f:
            f.write(midi_bytes)
            midi_path = f.name

        try:
            # music21 加载 MIDI
            score = converter.parse(midi_path, format="midi")

            # 智能推算: 调号
            score = self._infer_key_signature(score)

            # 智能推算: 拍号
            score = self._infer_time_signature(score)

            # 合并多音轨为一个谱表 (可选)
            # score = self._merge_parts(score)

            # 导出 MusicXML
            xml_path = midi_path.replace(".mid", ".xml")
            score.write("musicxml", fp=xml_path)

            with open(xml_path, "r", encoding="utf-8") as f:
                return f.read()

        finally:
            import os
            os.unlink(midi_path)
            if os.path.exists(xml_path := midi_path.replace(".mid", ".xml")):
                os.unlink(xml_path)

    def _infer_key_signature(self, score: stream.Score) -> stream.Score:
        """推算乐曲调号。基于 music21 的 Key 分析算法。"""
        try:
            analyzed_key = score.analyze("key")
            ks = key.Key(analyzed_key.tonic.name, analyzed_key.mode)
            for part in score.parts:
                part.insert(0, ks)
        except Exception:
            pass  # 推算失败则无调号
        return score

    def _infer_time_signature(self, score: stream.Score) -> stream.Score:
        """推算拍号。分析音符长度的模式。"""
        try:
            analyzed_meter = score.analyze("meter")
            ts = meter.TimeSignature(analyzed_meter)
            for part in score.parts:
                part.insert(0, ts)
        except Exception:
            pass
        return score

    def _merge_parts(self, score: stream.Score) -> stream.Score:
        """将多声部合并为单谱表 (钢琴谱: 高音+低音)"""
        # music21 的 stream.makeNotation 支持自动合并
        return score.makeNotation()
```

### 3.2 MusicXML → VexFlow JSON 预处理

```python
# app/algorithms/score/preprocessor.py
from lxml import etree


class VexFlowPreprocessor:
    """从 MusicXML 提取 VexFlow 友好的结构化数据。
    减少前端解析成本。"""

    def extract_measures(self, musicxml: str) -> list[dict]:
        """提取每个小节的数据结构。"""
        root = etree.fromstring(musicxml.encode())
        ns = {"xmlns": "http://www.example.com/musicxml"}  # 实际 XML 可能有 namespace

        measures = []
        for part in root.findall(".//part"):
            for measure_elem in part.findall("measure"):
                measure = {
                    "number": int(measure_elem.get("number", 0)),
                    "clef": self._extract_clef(measure_elem),
                    "time_sig": self._extract_time_sig(measure_elem),
                    "notes": self._extract_notes(measure_elem),
                }
                measures.append(measure)

        return measures

    def _extract_clef(self, measure):
        clef = measure.find(".//clef")
        if clef is not None:
            sign = clef.findtext("sign", "G")
            line = int(clef.findtext("line", "2"))
            return f"{sign}{line}"  # "G2", "F4"
        return "G2"

    def _extract_time_sig(self, measure):
        time = measure.find(".//time")
        if time is not None:
            beats = time.findtext("beats", "4")
            beat_type = time.findtext("beat-type", "4")
            return f"{beats}/{beat_type}"
        return None

    def _extract_notes(self, measure):
        notes = []
        for note_elem in measure.findall("note"):
            # 音高
            pitch = note_elem.find("pitch")
            if pitch is not None:
                step = pitch.findtext("step", "C")
                octave = pitch.findtext("octave", "4")
                alter = pitch.findtext("alter", "0")
                notes.append({
                    "pitch": f"{step}{octave}",
                    "accidental": "#" if alter == "1" else "b" if alter == "-1" else None,
                    "duration": int(note_elem.findtext("duration", "1")),
                    "type": note_elem.findtext("type", "quarter"),
                    "rest": False,
                })
            else:
                rest = note_elem.find("rest")
                if rest is not None:
                    notes.append({
                        "pitch": None,
                        "duration": int(note_elem.findtext("duration", "1")),
                        "type": note_elem.findtext("type", "quarter"),
                        "rest": True,
                    })
        return notes
```

### 4.1 VexFlow 封装

```typescript
// shared/src/features/score/VexFlowCanvas.tsx
import { useRef, useEffect, useImperativeHandle, forwardRef } from 'react';
import { Factory, Renderer, Stave, StaveNote, Voice, Formatter, Accidental } from 'vexflow';

interface VexFlowCanvasProps {
  /** MusicXML 中的小节数据 (或预处理的 VexFlow JSON) */
  measures: MeasureData[];
  /** 渲染宽度 (px) */
  width?: number;
  /** 渲染高度 (px) */
  height?: number;
  /** 当前播放到的小节索引 */
  currentMeasure?: number;
  /** 缩放比例 */
  scale?: number;

  onRenderComplete?: () => void;
}

export interface VexFlowCanvasHandle {
  /** 导出为 SVG 文本 */
  toSVG(): string;
  /** 导出为 Data URL (PNG) */
  toPNG(): Promise<string>;
  /** 滚动到指定小节 */
  scrollToMeasure(index: number): void;
}

export const VexFlowCanvas = forwardRef<VexFlowCanvasHandle, VexFlowCanvasProps>(
  ({ measures, width = 900, height = 600, currentMeasure, scale = 1 }, ref) => {
    const containerRef = useRef<HTMLDivElement>(null);
    const rendererRef = useRef<Renderer | null>(null);

    useEffect(() => {
      if (!containerRef.current || measures.length === 0) return;

      const container = containerRef.current;
      container.innerHTML = ''; // 清空

      const renderer = new Renderer(container, Renderer.Backends.CANVAS);
      renderer.resize(width, height);
      const ctx = renderer.getContext();
      ctx.scale(scale, scale);

      // 布局参数
      const staveWidth = width / scale - 100;
      const staveHeight = 120;
      const stavesPerPage = Math.floor(height / scale / staveHeight);

      // 逐小节渲染
      let currentX = 20;
      let currentY = 40;
      let measureIndex = 0;

      // 分组：每行显示多个小节
      const measuresPerStave = 4; // 每行 4 小节

      measures.forEach((measure, idx) => {
        // 换行
        if (idx > 0 && idx % measuresPerStave === 0) {
          currentX = 20;
          currentY += staveHeight;
        }

        // 每行第一小节画谱号
        const isFirstInRow = idx % measuresPerStave === 0;
        const stave = new Stave(currentX, currentY, staveWidth / measuresPerStave);

        if (isFirstInRow) {
          stave.addClef('treble');
          stave.addTimeSignature(measure.time_sig ?? '4/4');
        }

        stave.setContext(ctx).draw();

        // 高亮当前播放小节
        if (idx === currentMeasure) {
          ctx.fillStyle = 'rgba(255, 255, 0, 0.15)';
          ctx.fillRect(currentX, currentY - 10,
            staveWidth / measuresPerStave, staveHeight - 20);
          ctx.fillStyle = 'black';
        }

        // 添加音符
        if (measure.notes.length > 0) {
          try {
            const staveNotes = measure.notes.map((n) => {
              if (n.rest) {
                return new StaveNote({
                  keys: ['b/4'],
                  duration: n.type === 'quarter' ? 'qr' : n.type + 'r',
                });
              }
              const note = new StaveNote({
                keys: [n.pitch ?? 'c/4'],
                duration: n.type,
              });
              if (n.accidental) {
                note.addModifier(new Accidental(n.accidental));
              }
              return note;
            });

            const voice = new Voice({ num_beats: 4, beat_value: 4 });
            voice.addTickables(staveNotes);

            new Formatter()
              .joinVoices([voice])
              .format([voice], staveWidth / measuresPerStave - 10);

            voice.draw(ctx, stave);
          } catch (e) {
            console.warn(`Measure ${idx} rendering error:`, e);
          }
        }

        currentX += staveWidth / measuresPerStave;
      });

      rendererRef.current = renderer;
      onRenderComplete?.();

      return () => {
        rendererRef.current = null;
      };
    }, [measures, width, height, scale]);

    useImperativeHandle(ref, () => ({
      toSVG: () => {
        // 如果使用 SVG backend，直接返回 outerHTML
        return containerRef.current?.innerHTML ?? '';
      },
      toPNG: async () => {
        const canvas = containerRef.current?.querySelector('canvas');
        if (!canvas) return '';
        return canvas.toDataURL('image/png');
      },
      scrollToMeasure: (index) => {
        // 计算并滚动到目标小节
        const staveElements = containerRef.current?.querySelectorAll('.vf-stave');
        staveElements?.[index]?.scrollIntoView({ behavior: 'smooth' });
      },
    }));

    return (
      <div
        ref={containerRef}
        className="vexflow-container overflow-auto bg-white rounded-lg shadow-inner"
        style={{ width, height }}
      />
    );
  }
);

// ==== 类型 ====
export interface MeasureData {
  number: number;
  clef: string;
  time_sig: string | null;
  notes: NoteData[];
}

export interface NoteData {
  pitch: string | null;   // "C4", "E5", null for rest
  accidental: string | null;  // "#", "b", null
  duration: number;
  type: string;           // "whole", "half", "quarter", "eighth", "16th"
  rest: boolean;
}
```

```typescript
// shared/src/features/score/PianoRoll.tsx
interface PianoRollProps {
  notes: { pitch: number; start: number; end: number; velocity: number }[];
  width: number;
  height: number;
  currentTime?: number;
}

export function PianoRoll({ notes, width, height, currentTime }: PianoRollProps) {
  const canvasRef = useRef<HTMLCanvasElement>(null);

  useEffect(() => {
    const canvas = canvasRef.current;
    if (!canvas) return;
    const ctx = canvas.getContext('2d');
    if (!ctx) return;

    const dpr = window.devicePixelRatio || 1;
    canvas.width = width * dpr;
    canvas.height = height * dpr;
    ctx.scale(dpr, dpr);

    // 清空
    ctx.fillStyle = '#1a1a2e';
    ctx.fillRect(0, 0, width, height);

    // 琴键背景
    const keyHeight = height / 128;
    for (let p = 0; p < 128; p++) {
      const isWhite = [0, 2, 4, 5, 7, 9, 11].includes(p % 12);
      ctx.fillStyle = isWhite ? '#2a2a3e' : '#1a1a2e';
      ctx.fillRect(0, height - (p + 1) * keyHeight, width, keyHeight);
    }

    // 音符矩形
    const maxTime = Math.max(...notes.map((n) => n.end), 1);
    const timeScale = width / maxTime;

    notes.forEach((n) => {
      const x = n.start * timeScale;
      const y = height - (n.pitch + 1) * keyHeight;
      const w = (n.end - n.start) * timeScale;
      const h = keyHeight;

      ctx.fillStyle = `hsl(${n.pitch * 7}, 70%, 60%)`;
      ctx.fillRect(x, y, w, h);
      ctx.strokeStyle = 'rgba(255,255,255,0.3)';
      ctx.strokeRect(x, y, w, h);
    });

    // 播放进度线
    if (currentTime !== undefined) {
      ctx.strokeStyle = '#ff4444';
      ctx.lineWidth = 2;
      ctx.beginPath();
      ctx.moveTo(currentTime * timeScale, 0);
      ctx.lineTo(currentTime * timeScale, height);
      ctx.stroke();
    }
  }, [notes, width, height, currentTime]);

  return (
    <canvas ref={canvasRef} style={{ width, height }}
            className="rounded-lg" />
  );
}
```

## 5. 播放跟随系统

```
乐谱渲染 (VexFlow)
    │
    ▼
音频播放器 (Web Audio API / HTMLAudioElement)
    │
    ├── 每帧 (requestAnimationFrame) 读取 AudioContext.currentTime
    │   │
    │   ▼
    ├── 计算当前属于哪一小节
    │   │  measureIndex = binarySearch(measures, playHeadTime)
    │   │
    │   ▼
    └── 高亮当前小节 (currentMeasure prop → VexFlowCanvas)
        │
        ▼
    自动滚页 (scrollToMeasure)
        │
        ▼
    可选: 单音符高亮 (播放到的具体音符变色)
```

```typescript
// shared/src/hooks/useScorePlayer.ts
export function useScorePlayer(measures: MeasureData[], bpm: number) {
  const [currentMeasure, setCurrentMeasure] = useState(0);
  const [isPlaying, setIsPlaying] = useState(false);
  const audioRef = useRef<HTMLAudioElement | null>(null);
  const rafRef = useRef<number>(0);

  // 计算每个小节的起始时间 (秒)
  const measureTimes = useMemo(() => {
    const beatsPerMeasure = 4; // 默认 4/4
    const beatDuration = 60 / bpm;
    return measures.map((_, i) => i * beatsPerMeasure * beatDuration);
  }, [measures, bpm]);

  const play = useCallback((audioUrl: string) => {
    const audio = new Audio(audioUrl);
    audio.play();
    audioRef.current = audio;
    setIsPlaying(true);

    const update = () => {
      const currentTime = audio.currentTime;
      // 二分查找当前小节
      let low = 0, high = measureTimes.length - 1;
      while (low < high) {
        const mid = Math.floor((low + high + 1) / 2);
        if (measureTimes[mid] <= currentTime) {
          low = mid;
        } else {
          high = mid - 1;
        }
      }
      setCurrentMeasure(low);
      rafRef.current = requestAnimationFrame(update);
    };
    rafRef.current = requestAnimationFrame(update);

    audio.onended = () => {
      setIsPlaying(false);
      cancelAnimationFrame(rafRef.current);
    };
  }, [measureTimes]);

  const stop = useCallback(() => {
    audioRef.current?.pause();
    audioRef.current = null;
    setIsPlaying(false);
    cancelAnimationFrame(rafRef.current);
    setCurrentMeasure(0);
  }, []);

  useEffect(() => () => {
    audioRef.current?.pause();
    cancelAnimationFrame(rafRef.current);
  }, []);

  return { currentMeasure, isPlaying, play, stop };
}
```

## 7. 调号分析策略

```
music21.analysis.discrete.KrumhanslKeyAnalysis
  ─── 基于 Krumhansl-Schmuckler key-finding 算法
  ─── 分析音符分布 → 匹配调性模板 → 返回最可能的调号

算法输入: 量化后的 Note[] (MIDI pitch + duration)
算法输出: (tonic: str, mode: "major" | "minor")

示例:
  C Major:  tonic=C, mode=major
  A Minor:  tonic=A, mode=minor
  G Major:  tonic=G, mode=major
```

```python
def analyze_key(notes: list[Note]) -> tuple[str, str]:
    """不使用 music21 的轻量调性分析。"""
    # Krumhansl-Schmuckler 调性分布模板
    MAJOR_PROFILE = [6.35, 2.23, 3.48, 2.33, 4.38, 4.09, 2.52, 5.19, 2.39, 3.66, 2.29, 2.88]
    MINOR_PROFILE = [6.33, 2.68, 3.52, 5.38, 2.60, 3.53, 2.54, 4.75, 3.98, 2.69, 3.34, 3.17]

    # 统计各音级的加权出现次数
    weights = [0] * 12
    for note in notes:
        if not note.is_rest:
            weights[note.pitch_midi % 12] += note.duration_sec

    # 对 12 个调性计算相关系数
    best_score = -float("inf")
    best_key = ("C", "major")

    for tonic in range(12):
        for mode, profile in [("major", MAJOR_PROFILE), ("minor", MINOR_PROFILE)]:
            rotated = profile[-tonic:] + profile[:-tonic]
            score = sum(w * p for w, p in zip(weights, rotated))

            if score > best_score:
                best_score = score
                note_names = ["C", "C#", "D", "D#", "E", "F", "F#", "G", "G#", "A", "A#", "B"]
                best_key = (note_names[tonic], mode)

    return best_key
```
